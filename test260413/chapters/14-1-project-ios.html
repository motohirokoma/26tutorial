## 命名

- メンバ・関数: lowerCamelCase

- Models フォルダ: @Model や struct
- Services フォルダ: API/Auth/Sync

Step 3 — ApiClient (URLSession + async/await)

// Services/ApiClient.swift
import Foundation

struct ApiTask: Codable, Identifiable {
    let id: String
    let title: String
    let done: Bool
    let dueAt: Date?
    let priority: Int
    let updatedAt: Date
}

actor ApiClient {
    static let shared = ApiClient()
    private let base = URL(string: "https://tasq.example.com")!
    var sessionToken: String?  // Keychainから注入

    func fetchTasks(view: String) async throws -> [ApiTask] {
        var req = URLRequest(url: base.appending(path: "api/tasks").appending(queryItems: [.init(name: "view", value: view)]))
        req.setValue("Bearer \(sessionToken ?? "")", forHTTPHeaderField: "Authorization")
        let (data, res) = try await URLSession.shared.data(for: req)
        guard (res as? HTTPURLResponse)?.statusCode == 200 else { throw ApiError.badStatus }
        struct Wrap: Codable { let data: [ApiTask] }
        let dec = JSONDecoder(); dec.dateDecodingStrategy = .iso8601
        return try dec.decode(Wrap.self, from: data).data
    }
}

enum ApiError: Error { case badStatus, decode }
actor の効用

API client を actor にすると、データ競合 をコンパイラが検出してくれます。SwiftConcurrencyの恩恵を最大化。

同期サービス (差分マージ)

// Services/SyncService.swift
import SwiftData

@MainActor
final class SyncService {
    let context: ModelContext
    init(context: ModelContext) { self.context = context }

    func pull(view: String) async throws {
        let remote = try await ApiClient.shared.fetchTasks(view: view)
        for r in remote {
            let predicate = #Predicate<Task> { $0.id == r.id }
            if let local = try context.fetch(FetchDescriptor(predicate: predicate)).first {
                if r.updatedAt > local.updatedAt {
                    local.title = r.title; local.done = r.done
                    local.dueAt = r.dueAt; local.priority = r.priority
                    local.updatedAt = r.updatedAt
                }
            } else {
                context.insert(Task(id: r.id, title: r.title, done: r.done,
                                    dueAt: r.dueAt, priority: r.priority,
                                    updatedAt: r.updatedAt))
            }
        }
        try context.save()
    }
}

Keychain ラッパ (最小)

// Services/Keychain.swift
import Security
import Foundation

enum Keychain {
    static func save(_ value: String, for key: String) {
        let data = Data(value.utf8)
        let q: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecValueData as String: data,
        ]
        SecItemDelete(q as CFDictionary)
        SecItemAdd(q as CFDictionary, nil)
    }
    static func load(_ key: String) -> String? {
        let q: [String: Any] = [
            kSecClass as String: kSecClassGenericPassword,
            kSecAttrAccount as String: key,
            kSecReturnData as String: true,
        ]
        var item: CFTypeRef?
        guard SecItemCopyMatching(q as CFDictionary, &item) == errSecSuccess,
              let data = item as? Data else { return nil }
        return String(data: data, encoding: .utf8)
    }
}

  1. Claudeに「Models/Task.swift に reminderAt: Date? を追加。init にも反映」と依頼
  2. 変更後にXcodeで Clean Build Folder
  3. SwiftDataはマイグレーションが必要 (オプショナル追加なら自動)
  4. Simulator で起動して、サンプルデータを書き込み → 再起動 → 残っているか確認

Exercise 2
応用

SyncService にエラーリトライとオフライン耐性を追加

ネットワーク不通時もアプリが落ちないように、SyncService.pull を改良してください。

  1. 失敗時に最大3回リトライ (指数バックオフ)
  2. 3回失敗なら静かに諦め、ローカルデータをそのまま表示
  3. 成功時は UserDefaults に lastSyncedAt を記録
  4. UIから「最終同期: 5分前」のように表示できるよう Published プロパティを公開
  5. SimulatorでWi-FiをOFFにして挙動確認
模範解答を見る 
@MainActor
final class SyncService: ObservableObject {
    let context: ModelContext
    @Published var lastSyncedAt: Date?
    init(context: ModelContext) { self.context = context
        if let t = UserDefaults.standard.object(forKey: "lastSync") as? Date { lastSyncedAt = t }
    }

    func pull(view: String) async {
        for attempt in 0..<3 {
            do {
                try await pullOnce(view: view)
                lastSyncedAt = .now
                UserDefaults.standard.set(lastSyncedAt, forKey: "lastSync")
                return
            } catch {
                let delay = UInt64(pow(2.0, Double(attempt)) * 1_000_000_000)
                try? await Task.sleep(nanoseconds: delay)
            }
        }
        // 諦めてローカル表示続行
    }

    private func pullOnce(view: String) async throws { /* 本章 pull の中身 */ }
}

テスト: Wi-Fi OFFにしてアプリを起動 → クラッシュせずローカルキャッシュが表示される / Wi-Fi ON で再同期する を目視確認。
